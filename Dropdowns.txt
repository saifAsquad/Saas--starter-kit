    <script>
        const statusBtn1 = document.querySelector(".payBtn");

        const statusText1 = document.querySelector(".status-text1");
        const dropdown1 = document.querySelector(".status_dropdown1");

        const statusValues1 = document.querySelectorAll(".status-value1")
        statusBtn1.addEventListener("click", ShowDropDown);

        function ShowDropDown() {
            dropdown1.classList.toggle("active");

            if (dropdown1.classList.contains("active")) {

                dropdown1.classList.add("border-slate-300");
                dropdown1.classList.add("bg-white");
                dropdown1.style.maxHeight = dropdown1.scrollHeight + "px "

            } else {


                dropdown1.classList.remove("border-slate-300");
                dropdown1.classList.remove("bg-white");
                dropdown1.style.maxHeight = null;

            }

        }

        statusValues1.forEach(e => {
            e.addEventListener("click", () => {
                e.classList.toggle('active')

                if (e.classList.contains("active")) {

                    e.children[0].checked = true;
                    statusText1.textContent = e.children[1].textContent;
                }
                statusValues1.forEach(y => {
                    if (y.classList.contains("active") && y != e) {
                        y.classList.toggle("active");
                        y.children[0].checked = false;
                    }
                })


            });

        })
    </script>
